```python
def solution(nums):

    from itertools import combinations

    sums = []
    for i in combinations(nums, 3):
        sums.append(sum(i))

    Primes = []
    for i in sums:
        p = True
        for j in range(2,round(i**(1/2)+1)):
            if i % j == 0:
                p = False

        if p ==True: Primes.append(i)


    return len(Primes)
```

solution(nums)

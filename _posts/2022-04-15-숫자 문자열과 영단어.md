```python
def solution(s):
    import re

    new = s
    dict = ['zero', 'one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight', 'nine']
    for num,char in enumerate(dict):
        p = re.compile(char)
        new = p.sub(str(num), new)

    return int(new)
```
